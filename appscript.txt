// ════════════════════════════════════════════════════════════════
//  MUST #7 — SERVER-SIDE SESSION TOKEN VERIFICATION
//  Call at the top of every write action in doGet().
//  Returns true if session is valid, false if rejected.
//  Skips verification if no token sent (legacy compatibility — 
//  but logs the unverified call for audit visibility).
// ════════════════════════════════════════════════════════════════
function _verifySession(params, actionName) {
  const userId = sanitize(params.userId || params.UserId || params.AdminName || "", 30);
  const token  = sanitize(params.sessionToken || params.token || "", 64);
  if (!token || !userId) {
    // No token provided — allow but log (legacy/direct API call)
    logAction(userId || "Unknown", "UNVERIFIED_CALL", "Action: " + actionName + " — no session token");
    return true; // permissive: don't break existing clients without token
  }
  try {
    const sheet   = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.USERS);
    const allData = sheet.getDataRange().getValues();
    const headers = allData[0];
    const tokenCol = headers.indexOf("SessionToken");
    if (tokenCol === -1) return true; // column not yet created — allow
    for (let i = 1; i < allData.length; i++) {
      const rowId = String(allData[i][headers.indexOf("UserId")] || allData[i][0] || "").trim();
      if (rowId && (rowId === userId || String(allData[i][headers.indexOf("Name")]||"") === userId)) {
        const stored = String(allData[i][tokenCol] || "").trim();
        if (!stored) return true; // empty token = legacy user, allow
        return stored === token;
      }
    }
  } catch(e) { return true; } // on error, allow (don't block legitimate users)
  return true; // user not found but allow (will fail at action level)
}
// ── Hash password (SHA-256, matches client-side sha256())
function hashPassword(pwd) {
  const raw = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, pwd);
  return raw.map(b => ('0' + (b & 0xFF).toString(16)).slice(-2)).join('');
}

// ── Generate unique ID (UUID-style)
function generateId() {
  return Utilities.getUuid().replace(/-/g, "").substring(0, 16);

// ── Sanitize a parameter value
function sanitize(val, maxLen) {
  if (val === undefined || val === null) return "";
  let s = String(val).trim();
  if (maxLen && s.length > maxLen) s = s.substring(0, maxLen);
  return s;
}

// ── Automatic backup (set up via Time-based trigger)
// Backs up to a dedicated "Mandir_Backups" folder in Drive root.
// Deletes backups older than 30 days. Emails admin on failure.
function autoBackupSystem() {
  try {
    const ss             = SpreadsheetApp.getActiveSpreadsheet();
    const rootFolder     = DriveApp.getRootFolder();
    const backupFolderName = "Mandir_Backups";

    // Find or create the dedicated backup folder
    let backupFolder;
    const folderIter = rootFolder.getFoldersByName(backupFolderName);
    if (folderIter.hasNext()) {
      backupFolder = folderIter.next();
    } else {
      backupFolder = rootFolder.createFolder(backupFolderName);
    }

    // Create today's backup
    const today      = Utilities.formatDate(new Date(), "GMT+5:30", "yyyy-MM-dd");
    const backupName = "Backup_Mandir_" + today;
    DriveApp.getFileById(ss.getId()).makeCopy(backupName, backupFolder);
    logAction("System", "autoBackup", "Backup created: " + backupName);

    // Delete backups older than 30 days
    const cutoff = new Date();
    cutoff.setDate(cutoff.getDate() - 30);
    const files = backupFolder.getFiles();
    while (files.hasNext()) {
      const f = files.next();
      if (f.getDateCreated() < cutoff) {
        f.setTrashed(true);
        logAction("System", "autoBackup", "Old backup deleted: " + f.getName());
      }
    }
  } catch(err) {
    // Email all admins on backup failure
    try {
      const admins = getSheetData(SHEETS.USERS).filter(function(u) {
        return String(u.Role || "").toLowerCase() === "admin" &&
               String(u.Email || "").trim() !== "";
      });
      admins.forEach(function(admin) {
        try {
          MailApp.sendEmail({
            to:       String(admin.Email).trim(),
            subject:  CFG.name + " — BACKUP FAILED " + Utilities.formatDate(new Date(), "GMT+5:30", "dd-MM-yyyy"),
            htmlBody: "<h3 style='color:red;'>Automatic Backup Failed</h3><p>Error: " + err.message + "</p><p>Please backup manually from Google Sheets → File → Download.</p>"
          });
        } catch(mailErr) {}
      });
    } catch(e2) {}
    logAction("System", "autoBackup_FAILED", err.message);
  }
}

// ── Read sheet into array of objects
function getSheetData(name) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name);
  if (!sheet) return [];
  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return [];
  const headers = data.shift();
  return data.map(r => {
    let o = {};
    headers.forEach((h, i) => o[h] = r[i]);
    return o;
  });
}

// ════════════════════════════════════════════════════════════════
//  IMPROVEMENT #3 — SEQUENTIAL RECEIPT ID SYSTEM
//  Format: MNR-YYYY-00001  (5-digit, e.g. MNR-2025-00042)
//  Counter stored in PropertiesService, resets each year.
//  LockService prevents duplicate IDs under concurrent requests.
// ════════════════════════════════════════════════════════════════
function generateReceiptId() {
  const lock = LockService.getScriptLock();
  let lockAcquired = false;
  try {
    lock.waitLock(10000); // wait up to 10s for exclusive access
    lockAcquired = true;

    const props      = PropertiesService.getScriptProperties();
    const year       = new Date().getFullYear();
    const yearKey    = "RECEIPT_YEAR";
    const counterKey = "RECEIPT_COUNTER";

    let storedYear = props.getProperty(yearKey);
    let counter    = parseInt(props.getProperty(counterKey) || "0", 10);

    // Reset counter when year changes
    if (String(storedYear) !== String(year)) {
      counter = 0;
      props.setProperty(yearKey, String(year));
    }

    counter++;
    props.setProperty(counterKey, String(counter));

    // 5-digit padding: MNR-2025-00001 (handles up to 99999 per year)
    const padded = String(counter).padStart(5, "0");
    return CFG.receiptPrefix + "-" + year + "-" + padded;
  } finally {
    if (lockAcquired) lock.releaseLock();
  }
}

// ════════════════════════════════════════════════════════════════
//  IMPROVEMENT #4 — EMAIL QUOTA SYSTEM
//  canSendEmail() and incrementEmailCount() are merged into one
//  atomic LockService-protected operation: tryConsumeEmailQuota().
//  This prevents two simultaneous requests both passing the quota
//  check and pushing the count over the daily limit.
// ════════════════════════════════════════════════════════════════

// ── Atomic check-and-increment: returns true if email can be sent
// ALWAYS use this instead of calling canSendEmail() + incrementEmailCount() separately.
function tryConsumeEmailQuota() {
  const lock = LockService.getScriptLock();
  let lockAcquired = false;
  try {
    lock.waitLock(8000);
    lockAcquired = true;

    const props    = PropertiesService.getScriptProperties();
    const today    = Utilities.formatDate(new Date(), "GMT+5:30", "yyyy-MM-dd");
    const dateKey  = "EMAIL_DATE";
    const countKey = "EMAIL_COUNT";

    let storedDate = props.getProperty(dateKey)  || "";
    let emailCount = parseInt(props.getProperty(countKey) || "0", 10);

    // Reset on new day
    if (storedDate !== today) {
      emailCount = 0;
      props.setProperty(dateKey, today);
    }

    if (emailCount >= CFG.emailDailyLimit) return false;

    // Increment atomically inside the lock
    props.setProperty(countKey, String(emailCount + 1));
    return true;
  } catch(e) {
    Logger.log("tryConsumeEmailQuota error: " + e.toString());
    return false; // on lock timeout or any error, refuse to send (safe default)
  } finally {
    if (lockAcquired) lock.releaseLock(); // only release if we actually acquired it
  }
}

// ── Legacy wrappers kept for backward compatibility with existing call sites.
// Both now delegate to tryConsumeEmailQuota() logic without breaking the interface.
// canSendEmail() is READ-ONLY (no increment) — for display/check purposes only.
function canSendEmail() {
  const props    = PropertiesService.getScriptProperties();
  const today    = Utilities.formatDate(new Date(), "GMT+5:30", "yyyy-MM-dd");
  const dateKey  = "EMAIL_DATE";
  const countKey = "EMAIL_COUNT";
  let storedDate = props.getProperty(dateKey)  || "";
  let emailCount = parseInt(props.getProperty(countKey) || "0", 10);
  if (storedDate !== today) emailCount = 0;
  return emailCount < CFG.emailDailyLimit;
}

// FIX #4: incrementEmailCount() renamed to _incrementEmailCount() to prevent accidental
// direct calls that would bypass the atomic quota guard. Use tryConsumeEmailQuota() instead.
// @deprecated — internal use only. Do NOT call this directly.
function _incrementEmailCount() {
  const props    = PropertiesService.getScriptProperties();
  const countKey = "EMAIL_COUNT";
  const current  = parseInt(props.getProperty(countKey) || "0", 10);
  props.setProperty(countKey, String(current + 1));
}

function getEmailQuotaStatus() {
  const props    = PropertiesService.getScriptProperties();
  const today    = Utilities.formatDate(new Date(), "GMT+5:30", "yyyy-MM-dd");
  const dateKey  = "EMAIL_DATE";
  const countKey = "EMAIL_COUNT";
  let storedDate = props.getProperty(dateKey) || "";
  let count      = parseInt(props.getProperty(countKey) || "0", 10);
  if (storedDate !== today) count = 0;
  return { used: count, limit: CFG.emailDailyLimit, remaining: CFG.emailDailyLimit - count };
}

// ════════════════════════════════════════════════════════════════
//  #8 — EMAIL AUTOMATION SETTINGS (read/write toggles)
//  Stored in SETTINGS sheet: key=auto_receipt/monthly_report, value=1/0
// ════════════════════════════════════════════════════════════════
function getEmailSettings() {
  const ss    = SpreadsheetApp.getActiveSpreadsheet();
  let sheet   = ss.getSheetByName(SHEETS.SETTINGS);
  // Default OFF — never silently re-enable a toggle the user turned off
  if (!sheet) return { auto_receipt: false, monthly_report: false };
  const data  = sheet.getDataRange().getValues();
  if (!data || data.length === 0) return { auto_receipt: false, monthly_report: false };
  // Build map — skip header row where col A is "Key"
  // FIX: Google Sheets stores 1/0 as Number type, coerce to String for reliable comparison
  const map = {};
  data.forEach(r => {
    const k = String(r[0]||"").trim();
    if (k && k.toLowerCase() !== "key") map[k] = String(r[1]||"").trim();
  });
  return {
    // FIX: === "1" exact string match — "!== 0" was always true for Number 1
    auto_receipt:   map["auto_receipt"]   === "1",
    monthly_report: map["monthly_report"] === "1"
  };
}

function saveEmailSettings(key, value) {
  const ss  = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEETS.SETTINGS);
  if (!sheet) {
    sheet = ss.insertSheet(SHEETS.SETTINGS);
  }
  const data = sheet.getDataRange().getValues();
  let found  = false;
  // Search ALL rows (i=0) — handle both header and no-header sheet layouts
  for (let i = 0; i < data.length; i++) {
    const k = String(data[i][0]||"").trim();
    if (k === key) {
      sheet.getRange(i+1, 2).setValue(String(value));
      sheet.getRange(i+1, 3).setValue(Utilities.formatDate(new Date(),"GMT+5:30","dd-MM-yyyy HH:mm:ss"));
      found = true; break;
    }
  }
  if (!found) {
    // Append new row — no header needed, just key/value/timestamp
    sheet.appendRow([key, value, Utilities.formatDate(new Date(),"GMT+5:30","dd-MM-yyyy HH:mm:ss")]);
  }
  SpreadsheetApp.flush();
  logAction("Admin", "saveEmailSettings", key + " = " + value);
  return { status: "ok" };
}

//  Set up a Time-based trigger: Last day of month → sendMonthlyEndReports
//  Each member gets ONLY their own data. Privacy guaranteed.

function sendMonthlyEndReports(overrideMonth, overrideYear) {
  const settings = getEmailSettings();
  if (!settings.monthly_report) {
    Logger.log("Monthly report is disabled. Skipping.");
    return { status: "ok", sent: 0, skipped: 0, message: "Disabled" };
  }
  return _doSendMonthlyReports(overrideMonth, overrideYear);
}

// Internal — also called by triggerMonthlyReport (manual admin trigger)
function _doSendMonthlyReports(overrideMonth, overrideYear) {
  const MONTHS = ["January","February","March","April","May","June","July","August","September","October","November","December"];
  const now    = new Date();
  const month  = overrideMonth || MONTHS[now.getMonth()];
  const year   = String(overrideYear || now.getFullYear());

  const users         = getSheetData(SHEETS.USERS);
  const contributions = getSheetData(SHEETS.CONTRIBUTIONS);
  const types         = getSheetData(SHEETS.TYPES);

  const activeMembers = users.filter(u =>
    String(u.Role||"").toLowerCase() !== "admin" &&
    String(u.Status||"Active").toLowerCase() !== "inactive" &&
    String(u.Email||"").trim() !== ""
  );

  let sent = 0, skipped = 0;

  // FIX: Process each member individually with a sleep between sends.
  // Multiple contributions per member caused large HTML → GAS timeout → "network error" in admin UI.
  // Solution: pre-build all email bodies, then send with 300ms sleep between each.
  // This keeps execution well under the 6-minute GAS limit and avoids MailApp rate errors.
  const emailQueue = [];
  activeMembers.forEach(member => {
    const myContribs = contributions.filter(c =>
      String(c.UserId) === String(member.UserId) &&
      String(c.ForMonth) === month &&
      String(c.Year) === year
    );
    if (!canSendEmail()) { skipped++; return; } // fast read-only pre-check to avoid building HTML for nothing; atomic check happens in send loop via tryConsumeEmailQuota()
    let htmlBody;
    if (myContribs.length > 0) {
      const totalAmt = myContribs.reduce((s,c) => s + Number(c.Amount||0), 0);
      // FIX: pass types array to receipt email builder — missing was causing undefined type names
      htmlBody = _buildMonthlyReceiptEmail(member, myContribs, types, month, year, totalAmt);
    } else {
      htmlBody = _buildReminderEmail(member, month, year);
    }
    emailQueue.push({ to: String(member.Email).trim(), htmlBody: htmlBody, name: member.Name });
  });

  emailQueue.forEach(function(item) {
    try {
      if (!tryConsumeEmailQuota()) { skipped++; return; }
      MailApp.sendEmail({
        to:       item.to,
        subject:  CFG.name + " — " + month + " " + year + " Monthly Update",
        htmlBody: item.htmlBody
      });
      sent++;
      // Small sleep between sends avoids MailApp quota bursts and prevents timeout
      Utilities.sleep(300);
    } catch(e) {
      Logger.log("Failed to send to " + item.to + ": " + e.toString());
      skipped++;
    }
  });

  logAction("System", "Monthly Report Sent", month + " " + year + " — sent:" + sent + ", skipped:" + skipped);
  const quota = getEmailQuotaStatus();
  return { status: "ok", sent: sent, skipped: skipped, quotaRemaining: quota.remaining };
}

// ════════════════════════════════════════════════════════════════
//  MAIN GET HANDLER
// ════════════════════════════════════════════════════════════════
function doGet(e) {
  const action   = sanitize(e.parameter.action, 50);
  const callback = sanitize(e.parameter.callback, 100) || "cb";
  let result = {};


  // ── MUST #7: Verify session token for all write/mutating actions
  const _WRITE_ACTIONS = ["addContribution","addExpense","deleteContribution","deleteExpense",
    "addUser","deleteUser","updateUser","addGoal","deleteGoal","updateGoal",
    "addType","deleteType","updateType","addOccasion","deleteOccasion","updateOccasion",
    "addExpenseType","deleteExpenseType","updateExpenseType",
    "addEvent","deleteEvent","updateEvent","addEventExpense",
    "approveUser","rejectUser","saveEmailSettings","saveAnnouncement","clearAnnouncement",
    "saveBroadcast","updateFeedbackStatus","deleteFeedback"];
  if (_WRITE_ACTIONS.indexOf(action) !== -1 && !_verifySession(e.parameter, action)) {
    return ContentService
      .createTextOutput(callback + "(" + JSON.stringify({status:"error",message:"Session expired. Please login again."}) + ")")
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }

  // ── LOGIN
  if (action === "login") {
    let users   = getSheetData(SHEETS.USERS);
    let mobile  = sanitize(e.parameter.mobile, 15);
    let pwdHash = sanitize(e.parameter.password, 128);
     const _rl = _checkLoginRate(mobile);
     if (_rl.blocked) {
       result = { status: "fail", message: _rl.message };
    } else {
      let user = users.find(u => String(u.Mobile) === mobile && String(u.Password) === pwdHash);
      // FIX #1: Check specific statuses BEFORE the catch-all "not Active" guard,
      // so pending/rejected/inactive users receive the correct error message.
      // Previously the first condition caught ALL non-Active users, making the
      // else-if branches for pending/rejected/inactive unreachable dead code.
      if (!user) {
        _recordLoginFail(mobile);
        result = { status: "fail" };
      } else if (String(user.Status||"").toLowerCase() === "pending") {
        result = { status: "pending", message: "Your registration is pending admin approval. You will receive an email once approved." };
      } else if (String(user.Status||"").toLowerCase() === "rejected") {
        result = { status: "error", message: "Your registration was not approved. Please contact the temple admin." };
      } else if (String(user.Status||"").toLowerCase() === "inactive") {
        result = { status: "error", message: "Account is deactivated. Please contact admin." };
      } else if (user.Status !== "Active") {
        _recordLoginFail(mobile);
        result = { status: "fail" };
      } else {
        _clearLoginFail(mobile);
        let safeUser = Object.assign({}, user);
        delete safeUser.Password;
        const deviceInfo = sanitize(e.parameter.deviceInfo || "", 200);
        logAction(
          safeUser.Name || "User",
          "Login",
          "Mobile: " + mobile + " | Role: " + (safeUser.Role || "User") + " | UserId: " + (safeUser.UserId || ""),
          deviceInfo
        );
        result = { status: "success", user: safeUser };
      }
    }
  }

  // ── SET SESSION TOKEN (token only — does NOT touch TokenExpiry column)
  else if (action === "setSessionToken") {
    const userId = sanitize(e.parameter.userId, 30);
    const token  = sanitize(e.parameter.token,  64);
    // NOTE: expiry param is accepted but intentionally ignored — we do NOT write
    // TokenExpiry to the sheet. Session expiry is managed client-side only.
    if (!userId || !token) {
      result = { status: "error", message: "Missing userId or token" };
    } else {
      const sheet   = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.USERS);
      const data    = sheet.getDataRange().getValues();
      const headers = data[0];
      // Ensure SessionToken column exists
      let tokenCol = headers.indexOf("SessionToken");
      if (tokenCol === -1) { tokenCol = headers.length; sheet.getRange(1, tokenCol + 1).setValue("SessionToken"); }
      let found = false;
      for (let i = 1; i < data.length; i++) {
        if (String(data[i][0]) === userId) {
          sheet.getRange(i + 1, tokenCol + 1).setValue(token);
          SpreadsheetApp.flush();
          found = true; break;
        }
      }
      result = found ? { status: "ok" } : { status: "error", message: "User not found" };
    }
  }

  // ── CHECK SESSION TOKEN
  else if (action === "checkSession") {
    const userId = sanitize(e.parameter.userId, 30);
    const token  = sanitize(e.parameter.token,  64);
    if (!userId || !token) { result = { valid: true }; }
    else {
      try {
        const sheet    = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.USERS);
        const allData  = sheet.getDataRange().getValues();
        const headers  = allData[0];
        const tokenCol = headers.indexOf("SessionToken");
        if (tokenCol === -1) { result = { valid: true }; }
        else {
          let isValid = true;
          for (let i = 1; i < allData.length; i++) {
            if (String(allData[i][0]) === userId) {
              const stored = String(allData[i][tokenCol] || "").trim();
              if (stored === "") { isValid = true; break; }  // no token = legacy session, allow
              if (stored !== token) { isValid = false; break; } // token mismatch = another device
              isValid = true; break;
            }
          }
          result = { valid: isValid };
        }
      } catch(err) { result = { valid: true }; }
    }
  }

  // ── CLEAR SESSION TOKEN
  else if (action === "clearSessionToken") {
    const userId = sanitize(e.parameter.userId, 30);
    if (userId) {
      try {
        const sheet    = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.USERS);
        const allData  = sheet.getDataRange().getValues();
        const headers  = allData[0];
        const tokenCol = headers.indexOf("SessionToken");
        if (tokenCol !== -1) {
          for (let i = 1; i < allData.length; i++) {
            if (String(allData[i][0]) === userId) {
              sheet.getRange(i + 1, tokenCol + 1).setValue("");
              SpreadsheetApp.flush(); break;
            }
          }
        }
      } catch(e) { /* silent */ }
    }
    result = { status: "ok" };
  }

  // ── READ DATA
  else if (action === "getUsers")         result = getSheetData(SHEETS.USERS);
  else if (action === "getTypes")         result = _cachedSheet(SHEETS.TYPES,         "c_types",    600);
  else if (action === "getExpenseTypes")  result = _cachedSheet(SHEETS.EXPENSE_TYPES, "c_exptypes", 600);
  else if (action === "getOccasions")     result = _cachedSheet(SHEETS.OCCASIONS,     "c_occasions",600);
  else if (action === "getContributions") result = getSheetData(SHEETS.CONTRIBUTIONS);
  else if (action === "getExpenses")      result = getSheetData(SHEETS.EXPENSES);
  else if (action === "getEventData")     { result = { events: getSheetData(SHEETS.EVENTS), eventExpenses: getSheetData(SHEETS.EVENT_EXPENSES) }; }
  else if (action === "addEvent")         { result = _addEvent(e.parameter); }
  else if (action === "updateEvent")      { result = _updateEvent(e.parameter); }
  else if (action === "deleteEvent")      { result = _deleteEvent(e.parameter); }
  else if (action === "addEventExpense")  { result = _addEventExpense(e.parameter); }
  else if (action === "getYearConfig")    result = getSheetData("YEAR_CONFIG");

  // ── IMPROVEMENT #1: SEND RECEIPT EMAIL ON DEMAND (from admin Email button)
  else if (action === "sendContribReceiptEmail") {
    const userId      = sanitize(e.parameter.userId,       30);
    const receiptId   = sanitize(e.parameter.receiptId,    30);
    const userName    = sanitize(e.parameter.userName,    100);
    const amount      = sanitize(e.parameter.amount,       20);
    const forMonth    = sanitize(e.parameter.forMonth,     20);
    const year        = sanitize(e.parameter.year,         10);
    const typeName    = sanitize(e.parameter.typeName,    100);
    const occasionName= sanitize(e.parameter.occasionName,100);
    const note        = sanitize(e.parameter.note,        500);
    const paymentDate = sanitize(e.parameter.paymentDate,  50);
    const paymentMode = sanitize(e.parameter.paymentMode,  30);

    // Look up email from USERS sheet
    const users = getSheetData(SHEETS.USERS);
    const donor = users.find(u => String(u.UserId) === userId);
    const toEmail = donor ? String(donor.Email||"").trim() : "";

    if (!toEmail) {
      result = { status: "no_email" };
    } else if (!tryConsumeEmailQuota()) {
      result = { status: "quota" };
    } else {
      const sent = sendReceiptEmail(toEmail, userName, receiptId, amount, forMonth, year, typeName, occasionName, note, paymentDate, paymentMode);
      result = { status: sent ? "sent" : "error" };
    }
  }

  // ── IMPROVEMENT #4: GET EMAIL QUOTA STATUS (for admin sidebar badge)
  else if (action === "getEmailQuota") {
    result = getEmailQuotaStatus();
  }

  // ── #8: EMAIL AUTOMATION SETTINGS
  else if (action === "getEmailSettings") {
    result = getEmailSettings();
  }
  else if (action === "saveEmailSettings") {
    // Admin passes: action=saveEmailSettings&key=auto_receipt&value=1
    const key   = sanitize(e.parameter.key   || "", 30);
    const value = sanitize(e.parameter.value || "0", 2);
    const allowed = ["auto_receipt", "monthly_report", "logo_email_url"];
    if (allowed.indexOf(key) !== -1) {
      result = saveEmailSettings(key, value);
    } else {
      result = { status: "error", message: "Unknown setting key: " + key };
    }
  }

  // ── #9: MANUAL TRIGGER — Monthly Report (fire-and-store pattern)
  // WHY: Google's CDN cuts JSONP responses after ~30s for long-running doGet calls.
  // FIX: Store result in PropertiesService keyed by jobKey, return immediately.
  //      Client polls getMonthlyReportStatus every 5s to read the result.
  else if (action === "triggerMonthlyReport") {
    const month  = sanitize(e.parameter.month,  20);
    const year   = sanitize(e.parameter.year,   10);
    const jobKey = sanitize(e.parameter.jobKey, 50) || ("job_" + Date.now());
    // Mark job as pending immediately so client knows it started
    try {
      PropertiesService.getScriptProperties().setProperty(
        "MREPORT_" + jobKey, JSON.stringify({ status: "pending" })
      );
    } catch(e) {}
    // Run the report synchronously (GAS executes doGet fully before returning)
    // The JSONP response may be cut off by CDN, but the job WILL complete.
    // Result is also stored so the poll endpoint can return it.
    let jobResult;
    try {
      jobResult = _doSendMonthlyReports(month || null, year || null);
    } catch(err) {
      jobResult = { status: "error", message: err.message };
    }
    // Store final result for polling
    try {
      PropertiesService.getScriptProperties().setProperty(
        "MREPORT_" + jobKey, JSON.stringify(jobResult)
      );
    } catch(e) {}
    result = jobResult;
  }

  // ── #9b: POLL for monthly report result
  else if (action === "getMonthlyReportStatus") {
    const jobKey = sanitize(e.parameter.jobKey, 50);
    if (!jobKey) {
      result = { status: "error", message: "Missing jobKey" };
    } else {
      try {
        const raw = PropertiesService.getScriptProperties().getProperty("MREPORT_" + jobKey);
        if (!raw) {
          result = { status: "pending" }; // job not started yet or key expired
        } else {
          result = JSON.parse(raw);
          // Clean up once read successfully (only if done)
          if (result.status !== "pending") {
            try { PropertiesService.getScriptProperties().deleteProperty("MREPORT_" + jobKey); } catch(e) {}
          }
        }
      } catch(e) {
        result = { status: "pending" };
      }
    }
  }

  // ── GOAL MANAGEMENT
  else if (action === "addGoal") {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.GOALS);
    if (!sheet) { result = { status: "error", message: "GOALS sheet not found" }; }
    else {
      const headers = sheet.getRange(1,1,1,sheet.getLastColumn()).getValues()[0];
      const row = headers.map(h => {
        if (h === "GoalId")       return sanitize(e.parameter.GoalId, 50) || generateId();
        if (h === "GoalName")     return sanitize(e.parameter.GoalName, 200);
        if (h === "TargetAmount") return Number(e.parameter.TargetAmount||0);
        if (h === "CurrentAmount")return Number(e.parameter.CurrentAmount||0);
        if (h === "Status")       return sanitize(e.parameter.Status, 20) || "Disabled";
        if (h === "CreatedAt")    return formatDate();
        return "";
      });
      sheet.appendRow(row);
      logAction("Admin", "addGoal", sanitize(e.parameter.GoalName, 100));
      result = { status: "success" };
    }
  }
  else if (action === "updateGoal") {
    const rows    = getSheetData(SHEETS.GOALS);
    const sheet   = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.GOALS);
    const headers = sheet.getRange(1,1,1,sheet.getLastColumn()).getValues()[0];
    const rid     = sanitize(e.parameter.GoalId, 50);
    const idx     = rows.findIndex(r => String(r.GoalId) === rid);
    if (idx === -1) { result = { status: "error", message: "Goal not found" }; }
    else {
      const sheetRow = idx + 2;
      headers.forEach((h,i) => {
        if (h === "GoalName")      sheet.getRange(sheetRow,i+1).setValue(sanitize(e.parameter.GoalName,200));
        if (h === "TargetAmount")  sheet.getRange(sheetRow,i+1).setValue(Number(e.parameter.TargetAmount||0));
        if (h === "CurrentAmount") sheet.getRange(sheetRow,i+1).setValue(Number(e.parameter.CurrentAmount||0));
        if (h === "Status")        sheet.getRange(sheetRow,i+1).setValue(sanitize(e.parameter.Status,20));
      });
      logAction("Admin", "updateGoal", rid);
      // M7 — check if goal just completed
      _checkGoalCompletion(rid);
      result = { status: "updated" };
    }
  }
  else if (action === "deleteGoal") {
    const rows  = getSheetData(SHEETS.GOALS);
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.GOALS);
    const rid   = sanitize(e.parameter.GoalId, 50);
    const idx   = rows.findIndex(r => String(r.GoalId) === rid);
    if (idx === -1) { result = { status: "error", message: "Goal not found" }; }

            result = { status: "success", receiptUrls: newJson, receiptUrl: firstUrl };
            break;
          }
        }
        if (!result || !result.status) result = { status: "error", message: "Expense not found." };
      } catch(err) {
        result = { status: "error", message: err.message };
      }
    }

    else if (action === "removeAllExpenseReceipts") {
      try {
        const expId     = sanitize(e.parameter.Id, 30);
        const adminName = sanitize(e.parameter.AdminName, 100) || "Admin";
        const sheet   = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.EXPENSES);
        const data2   = sheet.getDataRange().getValues();
        const hdrs    = data2[0];
        const rcptCol  = hdrs.indexOf("ReceiptURL");
        const urlsCol  = hdrs.indexOf("ReceiptURLs");
        // Parse URLs to trash from client-sent list
        let urlsToTrash = [];
        try { urlsToTrash = JSON.parse(sanitize(e.parameter.OldReceiptURLs, 5000) || "[]"); } catch(ex) {}
        urlsToTrash.forEach(function(u) {
          if (u && u.includes("lh3.googleusercontent.com/d/")) {
            try {
              const fid = u.split("/d/")[1].split("?")[0].trim();
              if (fid && fid.length > 10) DriveApp.getFileById(fid).setTrashed(true);
            } catch(driveErr) {}
          }
        });
        for (let i = 1; i < data2.length; i++) {
          if (String(data2[i][0]) === expId) {
            if (urlsCol !== -1) sheet.getRange(i + 1, urlsCol + 1).setValue("[]");
            if (rcptCol !== -1) sheet.getRange(i + 1, rcptCol + 1).setValue("");
            SpreadsheetApp.flush();
            break;
          }
        }
        logAction(adminName, "removeAllExpenseReceipts", "ExpenseId: " + expId);
        result = { status: "success" };
      } catch(err) {
        result = { status: "error", message: err.message };
      }
    }

  // ── HEALTH CHECK
  else if (action === "getHealthCheck") {
    try {
      const ss     = SpreadsheetApp.getActiveSpreadsheet();
      const props  = PropertiesService.getScriptProperties();
      const quota  = getEmailQuotaStatus();
      const checks = {};

      // Check all required sheets exist
      const requiredSheets = ["USERS","CONTRIBUTIONS","EXPENSES","GOALS","TYPES",
        "OCCASIONS","EXPENSE_TYPES","AUDIT_LOGS","SETTINGS","GALLERY",
        "ANNOUNCEMENT","BROADCAST","EVENTS","EVENT_EXPENSES","CONTRIBUTION_REQUESTS"];
      requiredSheets.forEach(function(name) {
        checks["sheet_" + name] = !!ss.getSheetByName(name);
      });

      // Receipt counter
      checks["receipt_counter"] = parseInt(props.getProperty("RECEIPT_COUNTER") || "0", 10);
      checks["receipt_year"]    = props.getProperty("RECEIPT_YEAR") || "—";

      // Email quota
      checks["email_used"]      = quota.used;
      checks["email_limit"]     = quota.limit;
      checks["email_remaining"] = quota.remaining;

      // Backup — find latest backup in Mandir_Backups folder
      let lastBackup = "Never";
      try {
        const folderIter = DriveApp.getRootFolder().getFoldersByName("Mandir_Backups");
        if (folderIter.hasNext()) {
          const backupFolder = folderIter.next();
          const files = backupFolder.getFiles();
          let latest = null;
          while (files.hasNext()) {
            const f = files.next();
            if (!latest || f.getDateCreated() > latest.getDateCreated()) latest = f;
          }
          if (latest) lastBackup = Utilities.formatDate(latest.getDateCreated(), "GMT+5:30", "dd-MM-yyyy HH:mm");
        }
      } catch(be) {}
      checks["last_backup"] = lastBackup;

      // CFG version
      checks["version"] = CFG.version || "—";

      result = { status: "ok", checks: checks };
    } catch(err) {
      result = { status: "error", message: err.message };
    }
  }

  // ── MONTHLY CONTRIBUTION REMINDER (trigger on 25th via Time-based trigger)
  else if (action === "sendMonthlyReminders") {
    result = sendMonthlyContributionReminders(e.parameter.month, e.parameter.year);
  }

  // ── M16: Send OTP to verify NEW mobile number before changing
  else if (action === "sendMobileChangeOTP") {
    const userId    = sanitize(e.parameter.userId, 30);
    const newMobile = sanitize(e.parameter.newMobile, 15);
    if (!newMobile || !/^[6-9]\d{9}$/.test(newMobile)) {
      result = { status: "error", message: "Invalid mobile number format." };
    } else {
      // Check if new mobile already taken
      const users = getSheetData(SHEETS.USERS);
      const taken = users.find(u => String(u.Mobile||"") === newMobile && String(u.UserId||"") !== userId);
      if (taken) {
        result = { status: "error", message: "This mobile number is already registered to another account." };

        if (!user || !user.Email) {

              <h2 style="color:#946c44;">Mobile Number Change Verification</h2>

            </div>`

  // ── M16: Verify OTP and update mobile number
  else if (action === "verifyMobileChangeOTP") {
    const userId    = sanitize(e.parameter.userId, 30);
    const userOtp   = sanitize(e.parameter.otp, 10);

      const saved = JSON.parse(raw);
      if (saved.otp !== userOtp) {
        result = { status: "error", message: "Incorrect OTP. Please try again." };
      } else {

        const sheet   = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.USERS);
        const allData = sheet.getDataRange().getValues();

        const mobileCol = headers.indexOf("Mobile");

          for (let i = 1; i < allData.length; i++) {
            if (String(allData[i][headers.indexOf("UserId")]||allData[i][0]) === userId) {
              sheet.getRange(i+1, mobileCol+1).setValue(saved.newMobile);
              SpreadsheetApp.flush();
              logAction("User", "Mobile Number Changed", "UserId: " + userId + " | New: " + saved.newMobile);
              found = true; break;

          }
          result = found ? {status:"success"} : {status:"error",message:"User not found."};
        }
      }

  // ── M4: User submits contribution request (payment proof)
  else if (action === "submitContributionRequest") {
    try {
      const userId  = sanitize(e.parameter.UserId, 30);
      const amount  = sanitize(e.parameter.Amount, 20);
      const mode    = sanitize(e.parameter.PaymentMode || "UPI", 30);
      const note    = sanitize(e.parameter.Note || "", 500);

      const mode    = sanitize(e.parameter.PaymentMode || "UPI", 30);

    } catch(err) { result = { status:"error", message:err.message }; }
  }

  // ── User: get their own contribution requests
  else if (action === "getUserContributionRequests") {
    try {
      const userId = sanitize(e.parameter.userId || e.parameter.UserId || "", 30);

  else if (action === "getContributionRequests") {

      else {
        const data2 = sheet.getDataRange().getValues();

  else if (action === "resolveContributionRequest") {

      const reqId    = sanitize(e.parameter.ReqId, 30);
      const status   = sanitize(e.parameter.Status, 20); // Approved or Rejected

            result = {status:"success"};
            break;
          }
        }
        // FIX #6: Use explicit found flag instead of checking result.status,
        // which could throw TypeError if result is still {} at this point.
        if (!result || result.status !== "success") result = {status:"error",message:"Request not found"};
      }
    } catch(err) { result = {status:"error",message:err.message}; }
  }

  // ── User: delete their own APPROVED or REJECTED contribution request
  else if (action === "deleteContributionRequest") {
    try {
      const reqId  = sanitize(e.parameter.ReqId, 30);
      const userId = sanitize(e.parameter.userId || e.parameter.UserId || "", 30);
      if (!reqId || !userId) { result = { status:"error", message:"ReqId and userId required." }; }
      else {
        const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("CONTRIBUTION_REQUESTS");
        if (!sheet) { result = { status:"error", message:"Sheet not found." }; }
        else {
          const data2 = sheet.getDataRange().getValues();
          const hdrs  = data2[0];
          const idCol     = hdrs.indexOf("ReqId");
          const userCol   = hdrs.indexOf("UserId");

          let found = false;
          for (let i = 1; i < data2.length; i++) {
            const rowReqId  = String(data2[i][idCol]   || data2[i][0] || "");
            const rowUserId = String(data2[i][userCol]  || "");

              logAction(userId, "DeleteContributionRequest", "ReqId: " + reqId + " Status: " + rowStatus);
              result = { status:"deleted" }; found = true; break;
            }
          }
          if (!found) result = { status:"error", message:"Request not found or not eligible for deletion." };
        }
      }
    } catch(err) { result = { status:"error", message:err.message }; }
  }

  // ── Y1: Send OTP for registration email verification
  else if (action === "sendRegistrationOTP") {
    const email = sanitize(e.parameter.email, 200);
    if (!email) {
      result = { status: "error", message: "Email is required." };
    } else if (!tryConsumeEmailQuota()) {
      result = { status: "error", message: "Email limit reached for today. Please try again tomorrow." };
    } else {
      const otp = Math.floor(100000 + Math.random() * 900000);
      const otpKey = "regOtp_" + email.toLowerCase().replace(/[^a-z0-9]/g,"_").substring(0,30);
      CacheService.getScriptCache().put(otpKey, String(otp), 600); // 10 min
      MailApp.sendEmail({
        to: email,
        subject: CFG.name + " — Email Verification Code",
        htmlBody: `<div style="font-family:sans-serif;padding:20px;border:1px solid #f7a01a;border-radius:8px;max-width:480px;">
          <h2 style="color:#946c44;">${CFG.name}</h2>
          <p style="color:#475569;">You are registering a new account. Please verify your email with this code:</p>
          <h1 style="color:#f7a01a;font-size:48px;letter-spacing:8px;margin:16px 0;">${otp}</h1>
          <p style="color:#64748b;font-size:13px;">This code expires in 10 minutes. Do not share it with anyone.</p>
          <p style="color:#94a3b8;font-size:11px;">${CFG.name} | ${CFG.address}</p>
        </div>`
      });
      result = { status: "success", otpKey: otpKey };
    }
  }

  // ── Y1: Verify registration OTP
  else if (action === "verifyRegistrationOTP") {
    const otpKey  = sanitize(e.parameter.otpKey, 80);
    const userOtp = sanitize(e.parameter.otp, 10);
    if (!otpKey || !userOtp) {
      result = { status: "error", message: "Missing OTP key or code." };
    } else {
      const saved = CacheService.getScriptCache().get(otpKey);
      if (!saved) {
        result = { status: "error", message: "OTP expired or not found. Please request a new code." };
      } else if (saved !== userOtp) {
        result = { status: "error", message: "Incorrect code. Please try again." };
      } else {
        CacheService.getScriptCache().remove(otpKey);
        result = { status: "success" };
      }
    }
  }

  else { result = { error: "Invalid action: " + action }; }

  return ContentService
    .createTextOutput(callback + "(" + JSON.stringify(result) + ")")
    .setMimeType(ContentService.MimeType.JAVASCRIPT);
}

function _loginCacheKey(mobile) {
  // Short key: "lfail_" + last 4 digits of mobile
  // Not truly secure but prevents raw mobile exposure in cache keys
  // and is sufficient for rate-limiting purposes
  return "lfail_" + String(mobile).slice(-4) + "_" + String(mobile).length;
}
 
function _checkLoginRate(mobile) {
  if (!mobile) return { blocked: false };
  try {
    const cache = CacheService.getScriptCache();
    const key   = _loginCacheKey(mobile);
    const raw   = cache.get(key);
    if (!raw) return { blocked: false };
 
    const rec = JSON.parse(raw);
    if (!rec || !rec.count) return { blocked: false };
 
    // Block after 10 server-side attempts within 15 minutes
    if (rec.count >= 10) {
      return {
        blocked: true,
        message: "Too many login attempts. Please wait 15 minutes before trying again."
      };
    }
    return { blocked: false };
  } catch(e) {
    return { blocked: false }; // on cache error, never block legitimate users
  }
}
 
function _recordLoginFail(mobile) {
  if (!mobile) return;
  try {
    const cache = CacheService.getScriptCache();
    const key   = _loginCacheKey(mobile);
    const raw   = cache.get(key);
    let rec     = raw ? JSON.parse(raw) : { count: 0 };
    rec.count   = (rec.count || 0) + 1;
    // TTL: 15 minutes (900 seconds) — resets window on each failure
    cache.put(key, JSON.stringify(rec), 900);
  } catch(e) { /* silent — never crash login on cache error */ }
}
 
function _clearLoginFail(mobile) {
  if (!mobile) return;
  try {
    CacheService.getScriptCache().remove(_loginCacheKey(mobile));
  } catch(e) {}
}

// ── Server-side sheet cache helper (10-min TTL via CacheService)
// Used for master data that rarely changes: Types, Occasions, ExpenseTypes.
// Cache key is short (c_types etc.) to stay under CacheService 250-char key limit.
// On any write that modifies these sheets, call _bustSheetCache(key) to invalidate.
function _cachedSheet(sheetName, cacheKey, ttlSeconds) {
  const cache = CacheService.getScriptCache();
  const hit   = cache.get(cacheKey);
  if (hit) {
    try { return JSON.parse(hit); } catch(e) { /* corrupt cache — fall through */ }
  }
  const fresh = getSheetData(sheetName);
  try { cache.put(cacheKey, JSON.stringify(fresh), ttlSeconds); } catch(e) { /* silent */ }
  return fresh;
}
 
// ── Bust a specific server-side cache key after a write
// Call this inside addType, deleteType, addOccasion etc. after the write completes.
function _bustSheetCache(cacheKey) {
  try { CacheService.getScriptCache().remove(cacheKey); } catch(e) {}
}

function getYearlySummary() {
  try {
    const contributions = getSheetData(SHEETS.CONTRIBUTIONS);
    const expenses      = getSheetData(SHEETS.EXPENSES);
    const yearConfigRaw = getSheetData("YEAR_CONFIG");
 
    // ── Collect all years present in contributions or expenses
    const yearSet = new Set();
    contributions.forEach(c => {
      const y = parseInt(c.Year, 10);
      if (!isNaN(y) && y > 2000) yearSet.add(y);
    });
    expenses.forEach(e => {
      const y = parseInt(e.Year, 10);
      if (!isNaN(y) && y > 2000) yearSet.add(y);
    });
 
    // Also include any years defined in YEAR_CONFIG even if no transactions yet
    yearConfigRaw.forEach(row => {
      const y = parseInt(row.Year || row.year, 10);
      if (!isNaN(y) && y > 2000) yearSet.add(y);
    });
 
    // Sort years ascending
    const years = Array.from(yearSet).sort((a, b) => a - b);
 
    // ── Build a lookup for opening balances from YEAR_CONFIG
    // Supports column names: OpeningBalance, Opening_Balance, opening_balance, Balance, balance
    const openingMap = {};
    yearConfigRaw.forEach(row => {
      const y = parseInt(row.Year || row.year, 10);
      if (isNaN(y)) return;
      const ob = Number(
        row.OpeningBalance  || row.Opening_Balance  ||
        row.opening_balance || row.Balance          ||
        row.balance         || 0
      );
      openingMap[y] = isNaN(ob) ? 0 : ob;
    });
 
    // PERF: Pre-aggregate contributions and expenses into Maps keyed by year.
    // This is O(n) instead of O(n * numYears) — avoids re-scanning all rows per year.
    const contribByYear = {};
    contributions.forEach(c => {
      const y = parseInt(c.Year, 10);
      if (!isNaN(y)) contribByYear[y] = (contribByYear[y] || 0) + Number(c.Amount || 0);
    });
    const expenseByYear = {};
    expenses.forEach(e => {
      const y = parseInt(e.Year, 10);
      if (!isNaN(y)) expenseByYear[y] = (expenseByYear[y] || 0) + Number(e.Amount || 0);
    });

    // ── Build summary rows with carry-forward chain
    const rows = [];
    let carryForward = 0; // running carry-forward from previous year
 
    years.forEach(year => {
      // Opening balance = YEAR_CONFIG value OR carry-forward from last year
      // YEAR_CONFIG takes priority (admin may have set an explicit opening)
      const configOpening = openingMap[year] || 0;
      // Use explicit config opening if set, else use carry-forward
      const openingBalance = configOpening > 0 ? configOpening : carryForward;
 
      // Use pre-aggregated maps — O(1) lookup per year
      const totalCollection = contribByYear[year] || 0;
      const totalExpense    = expenseByYear[year]  || 0;
 
      // Closing balance = opening + collection - expense
      const closingBalance = openingBalance + totalCollection - totalExpense;
 
      // This year's closing becomes next year's carry-forward
      carryForward = closingBalance;
 
      rows.push({
        year:            year,
        openingBalance:  openingBalance,
        totalCollection: totalCollection,
        totalExpense:    totalExpense,
        closingBalance:  closingBalance,
        carryForward:    closingBalance  // same value — explicit for UI clarity
      });
    });
 
    return { status: "ok", rows: rows };
 
  } catch(err) {
    return { status: "error", message: err.message, rows: [] };
  }
}

function updateUser(p) {
  // M3 GUARD: Deactivating a user (Status=Inactive) only changes the USERS sheet.
  // CONTRIBUTIONS, EXPENSES and all financial records are NEVER touched here.
  // This is by design — inactive members' historical records must remain visible
  // and counted in all admin reports.
  let sheet   = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.USERS);
  let data    = sheet.getDataRange().getValues();
  let headers = data[0];
  let userId  = sanitize(p.UserId, 30);
  let found   = false;
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][0]) === userId) {
      // PERF: Build a full row array and write it in ONE setValues() call
      // instead of one sheet.getRange().setValue() per column.
      // This reduces Sheets API round-trips from ~10 calls to 1.
      const existingPwd = data[i][headers.indexOf("Password")];
      let newPwd = existingPwd;
      if (p.Password && sanitize(p.Password, 128).length > 0) {
        let pwd = sanitize(p.Password, 128);
        newPwd = pwd.length === 64 ? pwd : hashPassword(pwd);
      }
      // Clone existing row so columns not listed here are unchanged
      const newRow = data[i].slice();
      function _setCol(headerName, value) {
        const col = headers.indexOf(headerName);
        if (col !== -1 && value !== undefined) newRow[col] = value;
      }
      _setCol("Name",          sanitize(p.Name,    100));
      _setCol("Mobile",        sanitize(p.Mobile,  15));
      _setCol("Role",          sanitize(p.Role,    20));
      _setCol("Password",      newPwd);
      _setCol("Email",         sanitize(p.Email,   200));
      _setCol("Status",        sanitize(p.Status,  20) || "Active");
      _setCol("PhotoURL",      sanitize(p.PhotoURL, 500) || "");
      if (p.Village       !== undefined) _setCol("Village",       sanitize(p.Village, 100));
      if (p.Address       !== undefined) _setCol("Address",       sanitize(p.Address, 300));
      if (p.MonthlyTarget !== undefined) _setCol("MonthlyTarget", Number(p.MonthlyTarget||0));
      // Single batched write — 1 API call instead of ~10
      sheet.getRange(i + 1, 1, 1, newRow.length).setValues([newRow]);
      SpreadsheetApp.flush();
      found = true; break;
    }
  }
  if (found) { logAction(sanitize(p.AdminName,100)||"System","Update User","UserId: "+userId); return {status:"updated"}; }
  return { status: "not found" };
}

    function _setColRow(h, val) {
      const c = headers.indexOf(h);
      if (c !== -1 && val !== undefined) sheet.getRange(row, c+1).setValue(val);
    }
    _setColRow("Amount",      sanitize(body.Amount,    20));

  }

}

      }

      return ContentService.createTextOutput(
        JSON.stringify({status:"success", receiptUrl: existingUrls[0] || receiptUrl, receiptUrls: newUrlsJson})

// ── Core upload helper — uploads base64 to a specific Drive folder by ID
function _uploadToFolder(base64Data, fileName, folderId) {
  const folder      = DriveApp.getFolderById(folderId);
  const contentType = base64Data.substring(5, base64Data.indexOf(';'));
  const bytes       = Utilities.base64Decode(base64Data.split(',')[1]);
  const blob        = Utilities.newBlob(bytes, contentType, fileName);
  const file        = folder.createFile(blob);
  file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
  return "https://drive.google.com/thumbnail?id=" + file.getId() + "&sz=w400";
}

// ════════════════════════════════════════════════════════════════
//  M1 — MONTHLY REMINDER TO UNPAID MEMBERS (run on 25th via trigger)
//  Also callable manually from admin via triggerMonthlyReminder action.
//  Only runs if monthly_reminder toggle is ON in email settings.
// ════════════════════════════════════════════════════════════════
function sendMonthlyReminders(overrideMonth, overrideYear, bypassToggle) {
  const settings = getEmailSettings();
  // Toggle bypassed when bypassToggle=true (manual admin trigger)
  if (!bypassToggle && !settings.monthly_report) {
    return { status: "ok", sent: 0, skipped: 0, message: "Monthly emails disabled in settings." };
  }

  const MONTHS = ["January","February","March","April","May","June","July","August","September","October","November","December"];
  const now    = new Date();
  const month  = overrideMonth || MONTHS[now.getMonth()];
  const year   = String(overrideYear || now.getFullYear());

  const users         = getSheetData(SHEETS.USERS);
  const contributions = getSheetData(SHEETS.CONTRIBUTIONS);

  // Find active members (non-admin, active, has email) with NO contribution this month
  const unpaid = users.filter(function(u) {
    if (String(u.Role   || "").toLowerCase() === "admin")    return false;
    if (String(u.Status || "Active").toLowerCase() !== "active") return false;
    if (!String(u.Email || "").trim()) return false;
    // Check if they have any contribution for this month+year
    const paid = contributions.some(function(c) {
      return String(c.UserId) === String(u.UserId) &&
             String(c.ForMonth) === month &&
             String(c.Year) === year;
    });
    return !paid;
  });

//  Call this inside updateGoal or addContribution when CurrentAmount
//  reaches or exceeds TargetAmount. Marks goal "Completed" and emails admin.

function _checkGoalCompletion(goalId) {
  try {
    const goals   = getSheetData(SHEETS.GOALS);
    const goal    = goals.find(function(g) { return String(g.GoalId) === String(goalId); });

    const current = Number(goal.CurrentAmount || goal.Current || 0);
    const target  = Number(goal.TargetAmount  || goal.Target  || 0);
    const status  = String(goal.Status || "").toLowerCase();

    // Only fire if just reached/exceeded target and not already completed

      // Mark goal as Completed in sheet
      const sheet   = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.GOALS);

// ════════════════════════════════════════════════════════════════
//  MONTHLY CONTRIBUTION REMINDER (M1)
//  Call via Time-based trigger on 25th of each month.
//  Identifies members with no contribution recorded for current
//  month and sends a polite reminder email.
//  Admin can toggle on/off via Email Automation settings.
// ════════════════════════════════════════════════════════════════
function sendMonthlyContributionReminders(overrideMonth, overrideYear) {
  try {
    const settings = getEmailSettings();
    if (!settings.monthly_report) {
      logAction("System", "monthlyReminder", "Skipped — monthly_report toggle is OFF");
      return { status: "ok", sent: 0, skipped: 0, message: "Disabled" };
    }

    const MONTHS = ["January","February","March","April","May","June",
                    "July","August","September","October","November","December"];
    const now   = new Date();
    const month = overrideMonth || MONTHS[now.getMonth()];
    const year  = String(overrideYear || now.getFullYear());

    const users         = getSheetData(SHEETS.USERS);
    const contributions = getSheetData(SHEETS.CONTRIBUTIONS);

    // Members who are active, not admin, have an email
    const activeMembers = users.filter(function(u) {
      return String(u.Role   || "").toLowerCase() !== "admin" &&
             String(u.Status || "Active").toLowerCase() === "active" &&
             String(u.Email  || "").trim() !== "";
    });

    let sent = 0, skipped = 0;

    activeMembers.forEach(function(member) {
      // Check if this member has any contribution for current month+year
      const hasPaid = contributions.some(function(c) {
        return String(c.UserId)   === String(member.UserId) &&
               String(c.ForMonth) === month &&
               String(c.Year)     === year;
      });
      if (hasPaid) return; // already paid — no reminder needed

      if (!tryConsumeEmailQuota()) { skipped++; return; }
      try {
        MailApp.sendEmail({
          to:       String(member.Email).trim(),
          subject:  CFG.name + " — " + month + " " + year + " Contribution Reminder",
          htmlBody: _buildMonthlyReminderEmail(member, month, year)
        });
        sent++;
        Utilities.sleep(200);
      } catch(mailErr) {
        Logger.log("Reminder email failed for " + member.Email + ": " + mailErr);
        skipped++;
      }
    });

    logAction("System", "monthlyReminder", month + " " + year + " — sent:" + sent + ", skipped:" + skipped);
    return { status: "ok", sent: sent, skipped: skipped };
  } catch(err) {
    logAction("System", "monthlyReminder_FAILED", err.message);
    return { status: "error", message: err.message };
  }
}

    const name    = sanitize(params.Name    || "", 100);
    const mobile  = sanitize(params.Mobile  || "", 15);
    const email   = sanitize(params.Email   || "", 200);
    const pwd     = sanitize(params.Password || "", 200); // SHA-256 hashed on client
    const village = sanitize(params.Village || "", 100);
    const address = sanitize(params.Address || "", 300);

      return { status: "error", message: "Name, Mobile, Email and Password are required." };
    }
 
    const existingUsers = getSheetData(SHEETS.USERS);
 
    // Check duplicate mobile
    const dupMobile = existingUsers.find(u => String(u.Mobile || "") === mobile);
    if (dupMobile) {
      const ds = String(dupMobile.Status || "").toLowerCase();
      if (ds === "pending")  return { status: "error", message: "A registration with this mobile is already pending approval." };
      if (ds === "active")   return { status: "error", message: "This mobile number is already registered. Please login directly." };

function approveUser(params) {
  try {
    const userId    = sanitize(params.UserId    || "", 30);
    const adminName = sanitize(params.AdminName || "Admin", 100);
    if (!userId) return { status: "error", message: "UserId required." };
 
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.USERS);
    if (!sheet) return { status: "error", message: "Users sheet not found." };
 
    const data    = sheet.getDataRange().getValues();
    const headers = data[0];
    const statusIdx = headers.indexOf("Status");
    if (statusIdx === -1) return { status: "error", message: "Status column not found." };
 
    let userRow = -1;
    let userObj = {};
    for (let i = 1; i < data.length; i++) {
      if (String(data[i][headers.indexOf("UserId")] || "") === userId) {
        userRow = i + 1;
        headers.forEach((h, j) => { userObj[h] = data[i][j]; });
        break;
      }
    }
    if (userRow === -1) return { status: "error", message: "User not found." };
    if (String(userObj.Status || "").toLowerCase() !== "pending") {
      return { status: "error", message: "User is not in Pending status." };
    }
 
    sheet.getRange(userRow, statusIdx + 1).setValue("Active");
    SpreadsheetApp.flush();
    logAction(adminName, "Approve User", "UserId: " + userId + " | Name: " + userObj.Name + " | Mobile: " + userObj.Mobile);
 
    // Send approval email
    const toEmail = String(userObj.Email || "").trim();
    if (toEmail && tryConsumeEmailQuota()) {
      try {
        MailApp.sendEmail({
          to: toEmail,
          subject: CFG.name + " — Your Registration Has Been Approved!",
          htmlBody: _buildApprovalEmail(userObj)
        });
      } catch(mailErr) { Logger.log("Approval email error: " + mailErr); }
    }
 
    return { status: "success" };
  } catch(e) { return { status: "error", message: e.message }; }
}
 
function rejectUser(params) {
  try {
    const userId    = sanitize(params.UserId    || "", 30);
    const reason    = sanitize(params.Reason    || "", 500);
    const adminName = sanitize(params.AdminName || "Admin", 100);
    if (!userId) return { status: "error", message: "UserId required." };
 
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.USERS);
    if (!sheet) return { status: "error", message: "Users sheet not found." };
 
    const data    = sheet.getDataRange().getValues();
    const headers = data[0];
 
    let userRow = -1;
    let userObj = {};
    for (let i = 1; i < data.length; i++) {
      if (String(data[i][headers.indexOf("UserId")] || "") === userId) {
        userRow = i + 1;
        headers.forEach((h, j) => { userObj[h] = data[i][j]; });
        break;
      }
    }
    if (userRow === -1) return { status: "error", message: "User not found." };

    // Send rejection email BEFORE deleting row (need user data)
    const toEmail = String(userObj.Email || "").trim();
    if (toEmail && tryConsumeEmailQuota()) {
      try {
        MailApp.sendEmail({
          to: toEmail,
          subject: CFG.name + " — Regarding Your Registration Request",
          htmlBody: _buildRejectionEmail(userObj, reason)
        });
      } catch(mailErr) { Logger.log("Rejection email error: " + mailErr); }
    }

    // DELETE the row from sheet
    sheet.deleteRow(userRow);
    SpreadsheetApp.flush();
    logAction(adminName, "Reject & Delete User", "UserId: " + userId + " | Name: " + userObj.Name + (reason ? " | Reason: " + reason : ""));
 
    return { status: "success" };
  } catch(e) { return { status: "error", message: e.message }; }
}

// ── Public stats — safe aggregated data only, no personal info
function getPublicStats() {
  try {
    const contributions = getSheetData(SHEETS.CONTRIBUTIONS);
    const expenses      = getSheetData(SHEETS.EXPENSES);
    const users         = getSheetData(SHEETS.USERS);
    const goals         = getSheetData(SHEETS.GOALS);
    const yearConfigRaw = getSheetData("YEAR_CONFIG");
 
    const curYear = new Date().getFullYear();
 
    // ── This year totals
    const thisYearCollection = contributions
      .filter(c => parseInt(c.Year,10) === curYear)
      .reduce((s,c) => s + Number(c.Amount||0), 0);
 
    const thisYearExpense = expenses
      .filter(e => parseInt(e.Year,10) === curYear)
      .reduce((s,e) => s + Number(e.Amount||0), 0);
 
    // ── Active member count (no Admin, no Inactive)
    const memberCount = users.filter(u =>
      String(u.Role||"").toLowerCase() !== "admin" &&
      String(u.Status||"Active").toLowerCase() === "active"
    ).length;
 
    // ── Year summary rows (reuse getYearlySummary logic)
    const summary = getYearlySummary();
    const yearRows = summary.rows || [];
 
    // ── Current balance = last year's closing balance
    const balance = yearRows.length > 0
      ? yearRows[yearRows.length - 1].closingBalance
      : 0;
 
    // ── Goals — return only Name, Target, Current, Status (no IDs)
    const safeGoals = goals.map(g => ({
      Name:    g.Name    || g.GoalName || "",
      Target:  Number(g.Target  || g.TargetAmount  || 0),
      Current: Number(g.Current || g.CurrentAmount || 0),
      Status:  g.Status || "Enabled"
    }));
 
    const updatedAt = Utilities.formatDate(new Date(), "GMT+5:30", "dd-MMM-yyyy hh:mm a");
 
    return {
      status:             "ok",
      thisYearCollection: thisYearCollection,
      thisYearExpense:    thisYearExpense,
      balance:            balance,
      memberCount:        memberCount,
      yearRows:           yearRows,
      goals:              safeGoals,
      updatedAt:          updatedAt
    };
 
  } catch(err) {
    return { status: "error", message: err.message };
  }
}

function _changePassword(p) {  // FIX #3: Moved to global scope (was accidentally indented)
    try {
      const userId     = sanitize(p.UserId,      30);
      const oldHashRaw = sanitize(p.OldPassword, 128);
      const newHashRaw = sanitize(p.NewPassword, 128);
 
      if (!userId || !oldHashRaw || !newHashRaw) {
        return { status: "error", message: "Missing required fields." };
      }
      // Both must be 64-char SHA-256 hashes (client always sends hashed values)
      if (oldHashRaw.length !== 64 || newHashRaw.length !== 64) {
        return { status: "error", message: "Invalid password format." };
      }
      // Cannot set same password
      if (oldHashRaw === newHashRaw) {
        return { status: "error", message: "New password must be different." };
      }
 
      const sheet   = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.USERS);
      const data    = sheet.getDataRange().getValues();
      const headers = data[0];
      const pwdCol  = headers.indexOf("Password");
      const idCol   = 0; // UserId is always first column
 
      if (pwdCol === -1) return { status: "error", message: "Password column not found." };
 
      for (let i = 1; i < data.length; i++) {
        if (String(data[i][idCol]) === userId) {
          const storedHash = String(data[i][pwdCol] || "").trim().toLowerCase();
          // Verify old password matches stored hash
          if (storedHash !== oldHashRaw.toLowerCase()) {
            return { status: "error", message: "Current password is incorrect." };
          }
          // Write new password hash
          sheet.getRange(i + 1, pwdCol + 1).setValue(newHashRaw.toLowerCase());
          SpreadsheetApp.flush();
          logAction("User", "Change Password", "UserId: " + userId);
          return { status: "success" };
        }
      }
      return { status: "error", message: "User not found." };
    } catch(err) {
      return { status: "error", message: err.message };
    }
  }
function _addEvent(p) {  // FIX #3: Moved to global scope (was accidentally indented)
    try {
      const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.EVENTS);
      if (!sheet) return { status:"error", message:"EVENTS sheet not found." };

        if (h === "EventName")   return sanitize(p.EventName, 200);
        if (h === "Category")    return sanitize(p.Category, 50);
        if (h === "StartDate")   return sanitize(p.StartDate, 20);
        if (h === "EndDate")     return sanitize(p.EndDate, 20);
        if (h === "Budget")      return Number(p.Budget || 0);
        if (h === "Status")      return sanitize(p.Status, 30) || "Upcoming";
        if (h === "Description") return sanitize(p.Description, 500);
        if (h === "CreatedAt")   return formatDate();
        if (h === "AdminName")   return sanitize(p.AdminName, 100);
        return "";
      });
      sheet.appendRow(row);
      logAction(sanitize(p.AdminName,100)||"Admin", "addEvent", sanitize(p.EventName,100));
      return { status:"success" };
    } catch(err) { return { status:"error", message:err.message }; }
  }
 
function _updateEvent(p) {
    try {
      const sheet   = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.EVENTS);
      const data    = sheet.getDataRange().getValues();
      const headers = data[0];
      const eventId = sanitize(p.EventId, 50);
      // FIX #2: Declare rowIndex with let before the loop so the helper
      // function closes over the correct value. Previously var rowIndex
      // was declared inside the loop body, causing hoisting issues where
      // the helper would read undefined on the first access.
      let rowIndex = -1;
      // Helper: write a value to the column matching the given header name
      function setECol(h, v) {
        const c = headers.indexOf(h);
        if (c !== -1 && rowIndex !== -1) sheet.getRange(rowIndex+1, c+1).setValue(v);
      }
      for (let i = 1; i < data.length; i++) {
        if (String(data[i][headers.indexOf("EventId")]) === eventId) {
          rowIndex = i;
          setECol("EventName",   sanitize(p.EventName,200));
          setECol("Category",    sanitize(p.Category,50));
          setECol("Status",      sanitize(p.Status,30));
          setECol("StartDate",   sanitize(p.StartDate,20));
          setECol("EndDate",     sanitize(p.EndDate,20));
          setECol("Budget",      Number(p.Budget||0));
          setECol("Description", sanitize(p.Description,500));
          SpreadsheetApp.flush();
          logAction(sanitize(p.AdminName,100)||"Admin","updateEvent",eventId);
          return { status:"updated" };
        }
      }
      return { status:"not found" };
    } catch(err) { return { status:"error", message:err.message }; }
  }
 
function _deleteEvent(p) {  // FIX #3: Moved to global scope (was accidentally indented)
    try {
      const eventId  = sanitize(p.EventId, 50);
      const adminName = sanitize(p.AdminName,100) || "Admin";
      // Delete event row
      const evSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.EVENTS);
      const evData  = evSheet.getDataRange().getValues();
      const evHdrs  = evData[0];
      for (let i = evData.length-1; i >= 1; i--) {
        if (String(evData[i][evHdrs.indexOf("EventId")]) === eventId) {
          evSheet.deleteRow(i+1); break;
        }
      }
      // Delete all linked event expenses
      const exSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.EVENT_EXPENSES);
      if (exSheet) {
        const exData = exSheet.getDataRange().getValues();
        const exHdrs = exData[0];
        const eidCol = exHdrs.indexOf("EventId");
        for (let i = exData.length-1; i >= 1; i--) {
          if (String(exData[i][eidCol]) === eventId) exSheet.deleteRow(i+1);
        }
      }
      SpreadsheetApp.flush();
      logAction(adminName, "deleteEvent", eventId);
      return { status:"deleted" };
    } catch(err) { return { status:"error", message:err.message }; }
  }
